name: \"content\"\n type: \"text\"\n required: true\n description: \"The full body content of the announcement.\"\n - name: \"published_at\"\n type: \"datetime\"\n required: true\ description: \"The date and time the announcement was published.\"\n - name: \"author\"\n type: \"string\"\n required: false\n constraints:\n max_length: 255\n description: \"The author or responsible party for the announcement (e.g., \'Headquarters\', \'Commander\'s Office\').\"\n - name: \"tags\"\n type: \"json\"\n required: false\n description: \"Keywords or categories for the announcement (e.g., \'policy\', \'safety\', \'recruitment\').\"\n - name: \"attachments\"\n type: \"json\"\n required: false\n description: \"An array of objects, each representing an attached file with its URL and optional metadata (e.g., [{\'url\': \'...\', \'title\': \'...\'}]).\"\n\naccess_patterns:\n - name: \"value\"\n type: \"text\"\n required: true\n description: \"The actual text content.\"\n - name: \"description\"\n type: \"text\"\n required: false\n description: \"A brief description of where and how this copy is used.\"\n\naccess_patterns:\n -
